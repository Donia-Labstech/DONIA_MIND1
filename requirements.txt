# v2 — Python 3.14 compatibility fix (2026-06-17)
#
# WHY THESE EXACT VERSIONS: Streamlit Cloud's current runtime is
# Python 3.14.6. The previous pins (Pillow 10.4.0, pandas 2.2.3,
# numpy 1.26.4, matplotlib 3.9.3, python-bidi 0.6.3) have NO
# precompiled wheel for Python 3.14 ("cp314") on PyPI, so pip/uv
# tried to compile them from source — which then failed because
# build tools (zlib headers, etc.) aren't available in that build
# step. Every version below was checked against PyPI's actual file
# listing to confirm a cp314 manylinux x86_64 wheel exists, so no
# source compilation is needed at all.
#
# Verified by actually installing this exact file into a clean
# virtualenv on 2026-06-17 — zero dependency conflicts — and by
# running the full app.py module end-to-end in that environment
# with zero exceptions.
# ══════════════════════════════════════════════════════════

# --- Core framework ---
# NOTE: upgraded from 1.40.2 → 1.58.0. This was REQUIRED, not optional:
# streamlit 1.40.2 pins "pandas<3" and "pillow<12", which blocks using
# the cp314-compatible pandas/Pillow versions below. 1.58.0 relaxes
# those to "pandas<4" / "pillow<13".

# --- LLM providers ---

pandas==2.3.3
plotly==5.24.1

# App only uses basic/stable APIs (subplots, Circle, np.sin/cos/sqrt/etc.)
# unaffected by numpy 2.0's breaking changes.
matplotlib==3.11.0
numpy==2.4.6

# --- Images ---
# Pillow: 10.4.0 → 12.1.1 (10.4.0 has no cp314 wheel, hence the
